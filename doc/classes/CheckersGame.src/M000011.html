        def ==(other)
                ROW_LOOKUP.keys.each do |row|
                        COLUMN_LOOKUP.keys.each do |column|
                                if(@gameboard[ROW_LOOKUP[row]][COLUMN_LOOKUP[column]]!=other[row,column])
                                        return false
                                end
                        end
                end
                return @whose_turn==other.whose_turn
        end
